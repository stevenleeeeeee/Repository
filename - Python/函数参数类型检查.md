```python
import inspect

def checkparams(fn):
    """ 对接收来的函数对象解析其函数签名然后判断其参数类型与实参是否对应 """
    def wapper(*args,**kwargs):             #接收函数参数
        sig = inspect.signature(fn)         #获取函数签名
        params = sig.parameters             #获取签名中的参数类型
        values = list(params.values())**    #字典的values，为的是通过values.annotation解放注释类型**

        for i,p in enumerate(args):
            param = values[i]
            if param.annotation is not param.empty and isinstance(p,values[i].annotation):
                print('OKKKK')
            else:
                print("参数类型不正确")

        for k,v in kwargs.items():
            if params[k].annotation is not params[k].empty and isinstance(v,params[k].annotation):
                print("OK")
            else:
                print("参数类型不正确")
        func = fn(*args,**kwargs)
        return func
    return wapper

@checkparams
def add(x,y:int=7)->int:
    return x+y
print(add(3,y=4))
```

#### Example2
```python
from inspect import signature
from functools import wraps

def typeassert(*ty_args, **ty_kwargs):
    def decorate(func):
        # If in optimized mode, disable type checking
        if not __debug__:
            return func

        # Map function argument names to supplied types
        sig = signature(func)
        bound_types = sig.bind_partial(*ty_args, **ty_kwargs).arguments

        @wraps(func)
        def wrapper(*args, **kwargs):
            bound_values = sig.bind(*args, **kwargs)
            # Enforce type assertions across supplied arguments
            for name, value in bound_values.arguments.items():
                if name in bound_types:
                    if not isinstance(value, bound_types[name]):
                        raise TypeError(
                            'Argument {} must be {}'.format(name, bound_types[name])
                            )
            return func(*args, **kwargs)
        return wrapper
    return decorate


# 这个装饰器非常灵活，既可指定所有参数类型，也可只指定部分。 并且可通过位置或关键字来指定参数类型。下面示例

>>> @typeassert(int, z=int)
... def spam(x, y, z=42):
...     print(x, y, z)
...
>>> spam(1, 2, 3)
1 2 3
>>> spam(1, 'hello', 3)
1 hello 3
>>> spam(1, 'hello', 'world')
```
