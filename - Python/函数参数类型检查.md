import inspect

def checkparams(fn):
    """ 对接收来的函数对象解析其函数签名然后判断其参数类型与实参是否对应 """
    def wapper(*args,**kwargs):             #接收函数参数
        sig = inspect.signature(fn)         #获取函数签名
        params = sig.parameters             #获取签名中的参数类型
        values = list(params.values())**    #字典的values，为的是通过values.annotation解放注释类型**

        for i,p in enumerate(args):
            param = values[i]
            if param.annotation is not param.empty and isinstance(p,values[i].annotation):
                print('OKKKK')
            else:
                print("参数类型不正确")

        for k,v in kwargs.items():
            if params[k].annotation is not params[k].empty and isinstance(v,params[k].annotation):
                print("OK")
            else:
                print("参数类型不正确")
        func = fn(*args,**kwargs)
        return func
    return wapper

@checkparams
def add(x,y:int=7)->int:
    return x+y
