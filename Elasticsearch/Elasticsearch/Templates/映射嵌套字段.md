```txt
PUT /my_index
{
  "mappings": {
    "blogpost": {
      "properties": {
        "comments": {
          "type": "nested",         # 设置字段为 nested 很简单: 只需要将字段类型 object 替换为 nested 即可
          "properties": {           # 至此所有 comments 对象会被索引在独立的嵌套文档
            "name":    { "type": "string"  },
            "comment": { "type": "string"  },
            "age":     { "type": "short"   },
            "stars":   { "type": "short"   },
            "date":    { "type": "date"    }
          }
        }
      }
    }
  }
}
```
